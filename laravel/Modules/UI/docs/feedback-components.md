### Animazioni

```scss
  from { opacity: 0; }
  to { opacity: 1; }
}

  from { opacity: 1; }
  to { opacity: 0; }
}

// Slide in/out
@keyframes slideIn {
  from { transform: translateY(100%); }
  to { transform: translateY(0); }
}

@keyframes slideOut {
  from { transform: translateY(0); }
  to { transform: translateY(100%); }
}
```
